BATCH CONVERSION 
20 May 2024

ref: https://askubuntu.com/a/35994 


$ find . -iname '*.jpg' -exec mogrify -format webp -quality 80 {} +

$ for f in $(find . -iname "*.webp"); do 
    mv "$f" "${f%.webp}.jpg.webp"
done


$ find . -iname '*.jpeg' -exec mogrify -format webp -quality 80 {} +

$ for f in $(find . -iname "*.webp" -and -not -iname "*.jpg.webp"); do
    mv "$f" "${f%.webp}.jpeg.webp"
done


$ find . -iname '*.png' -exec mogrify -format webp -quality 80 {} +

$ for f in $(find . -iname "*.webp" -and -not -iname "*.jpg.webp" -and -not -iname "*.jpeg.webp"); do
    mv "$f" "${f%.webp}.png.webp" 
done
